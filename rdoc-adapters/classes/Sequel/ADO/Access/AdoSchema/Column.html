def db_type
  t = DATA_TYPE[self["DATA_TYPE"]]
  if t == "DECIMAL" && precision
    t + "(#{precision.to_i},#{(scale || 0).to_i})"
  elsif t == "TEXT" && maximum_length && maximum_length > 0
    t + "(#{maximum_length.to_i})"
  else
    t
  end
end
