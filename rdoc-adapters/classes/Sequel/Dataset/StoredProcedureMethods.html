# File lib/sequel/adapters/utils/stored_procedures.rb, line 24
def run(&block)
  case @sproc_type
  when :select, :all
    all(&block)
  when :first
    first
  when :insert
    insert
  when :update
    update
  when :delete
    delete
  end
end
