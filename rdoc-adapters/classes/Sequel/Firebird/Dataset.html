def fetch_rows(sql)
  execute(sql) do |s|
    begin
      @columns = columns = s.fields.map{|c| output_identifier(c.name)}
      s.fetchall.each do |r|
        h = {}
        r.zip(columns).each{|v, c| h[c] = v}
        yield h
      end
    ensure
      s.close
    end
  end
  self
end
