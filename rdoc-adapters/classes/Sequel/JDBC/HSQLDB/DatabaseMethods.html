# File lib/sequel/adapters/jdbc/hsqldb.rb, line 26
def db_version
  @db_version ||= begin
    v = get{DATABASE_VERSION(){}}
    if v =~ /(\d+)\.(\d+)\.(\d+)/
      $1.to_i * 10000 + $2.to_i * 100 + $3.to_i
    end
  end
end
