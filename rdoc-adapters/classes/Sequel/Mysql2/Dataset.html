def fetch_rows(sql)
  execute(sql) do |r|
    @columns = if identifier_output_method
      r.fields.map!{|c| output_identifier(c.to_s)}
    else
      r.fields
    end
    r.each(:cast_booleans=>convert_tinyint_to_bool?){|h| yield h}
  end
  self
end
