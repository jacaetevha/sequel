def associate(type, name, opts = OPTS, &block)
  if block || opts[:dataset] || (opts[:clone] && association_reflection(opts[:clone])[:prepared_statement] == false)
    opts = opts.merge(:prepared_statement=>false)
  end
  super(type, name, opts, &block)
end
