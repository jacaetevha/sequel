def eager_loading_dataset(opts, ds, select, associations, eager_options=OPTS)
  ds = super(opts, ds, select, associations, eager_options)
  if !ds.opts[:server] and s = eager_options[:self] and server = s.opts[:server]
    ds = ds.server(server)
  end
  ds
end
