def get_pool(db, opts = OPTS)
  case v = connection_pool_class(opts)
  when Class
    v.new(db, opts)
  when Symbol
    require("sequel/connection_pool/#{v}")
    connection_pool_class(opts).new(db, opts) || raise(Sequel::Error, "No connection pool class found")
  end
end
