def reverse(&block)
  begin
    instance_eval(&block)
  rescue
    just_raise = true
  end
  if just_raise
    Proc.new{raise Sequel::Error, 'irreversible migration method used, you may need to write your own down method'}
  else
    actions = @actions.reverse
    Proc.new do
      actions.each do |a|
        if a.last.is_a?(Proc)
          pr = a.pop
          send(*a, &pr)
        else
          send(*a)
        end
      end
    end
  end
end
